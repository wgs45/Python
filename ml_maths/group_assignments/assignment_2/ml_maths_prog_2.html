import numpy as np

np.random.seed(42)

def generate_xor(n_samples=1000, noise=0.1):
    X = np.random.randint(0, 2, (n_samples, 2))
    y = (X[:, 0] ^ X[:, 1]).reshape(-1, 1)

    X = X + noise * np.random.randn(n_samples, 2)
    return X, y

def generate_quadrant(train_size=4000, test_size=1000):
    def create_data(n):
        X = np.random.uniform(-1, 1, (n, 2))
        y = np.zeros((n, 1))

        y[(X[:, 0] * X[:, 1]) > 0] = 1
        return X, y

    X_train, y_train = create_data(train_size)
    X_test, y_test = create_data(test_size)

    return X_train, y_train, X_test, y_test

if __name__ == "__main__":
    X_xor, y_xor = generate_xor()
    X_train, y_train, X_test, y_test = generate_quadrant()

    print("XOR:", X_xor.shape, y_xor.shape)
    print("Train:", X_train.shape, y_train.shape)
    print("Test:", X_test.shape, y_test.shape)

def sigmoid(z):
    return 1 / (1 + np.exp(-z))

def sigmoid_deriv(z):
    return sigmoid(z) * (1 - sigmoid(z))

def softmax(z):
    exp = np.exp(z - np.max(z))
    return exp / np.sum(exp, axis=1, keepdims=True)

def init_weights(input_dim, hidden_dim, output_dim):
    W1 = np.random.randn(input_dim, hidden_dim) * 0.01
    b1 = np.zeros((1, hidden_dim))
    W2 = np.random.randn(hidden_dim, output_dim) * 0.01
    b2 = np.zeros((1, output_dim))
    return W1, b1, W2, b2


def forward(X, W1, b1, W2, b2, use_softmax=False):
    Z1 = X @ W1 + b1
    A1 = sigmoid(Z1)

    Z2 = A1 @ W2 + b2
    if use_softmax:
        A2 = softmax(Z2)
    else:
        A2 = sigmoid(Z2)

    return Z1, A1, Z2, A2

def compute_loss(Y, A2):
    m = Y.shape[0]
    cost = -np.mean(Y * np.log(A2 + 1e-8) + (1 - Y) * np.log(1 - A2 + 1e-8))
    return cost

def backward(X, Y, Z1, A1, Z2, A2, W2):
    m = X.shape[0]

    dZ2 = A2 - Y
    dW2 = (1 / m) * (A1.T @ dZ2)
    db2 = (1 / m) * np.sum(dZ2, axis=0, keepdims=True)

    dZ1 = (dZ2 @ W2.T) * (A1 * (1 - A1)) # Sigmoid deriv
    dW1 = (1 / m) * (X.T @ dZ1)
    db1 = (1 / m) * np.sum(dZ1, axis=0, keepdims=True)

    return dW1, db1, dW2, db2

def update(W1, b1, W2, b2, dW1, db1, dW2, db2, lr=0.1):
    W1 -= lr * dW1
    b1 -= lr * db1
    W2 -= lr * dW2
    b2 -= lr * db2
    return W1, b1, W2, b2

import numpy as np

def relu(z):
    return np.maximum(0, z)

def forward_relu_hidden(X, W1, b1, W2, b2):
    Z1 = X @ W1 + b1
    A1 = relu(Z1)
    Z2 = A1 @ W2 + b2
    A2 = 1 / (1 + np.exp(-Z2))
    return Z1, A1, Z2, A2

X_train, y_train, X_test, y_test = generate_quadrant(train_size=4000, test_size=1000)

hidden_dim  = 6
epochs      = 2200
lr          = 0.01
batch_size  = 128
eps         = 1e-8

input_dim  = X_train.shape[1]   # =2
output_dim = y_train.shape[1]   # =1

W1 = np.random.randn(input_dim, hidden_dim) * np.sqrt(2 / input_dim)  #  He init
b1 = np.zeros((1, hidden_dim))
W2 = np.random.randn(hidden_dim, output_dim) * 0.01
b2 = np.zeros((1, output_dim))

train_loss_hist = []
test_loss_hist  = []

N = X_train.shape[0]

for epoch in range(epochs):
    # (a) shuffle
    idx = np.random.permutation(N)
    Xs = X_train[idx]
    ys = y_train[idx]

    # (b) mini-batch
    for start in range(0, N, batch_size):
        end = start + batch_size
        Xb = Xs[start:end]
        yb = ys[start:end]
        m  = Xb.shape[0]

        Z1, A1, Z2, A2 = forward_relu_hidden(Xb, W1, b1, W2, b2)

        # loss (BCE)
        A2_clip = np.clip(A2, eps, 1 - eps)
        loss = -np.mean(yb * np.log(A2_clip) + (1 - yb) * np.log(1 - A2_clip))

        # output layer (sigmoid + BCE) => dZ2 = A2 - y
        dZ2 = (A2 - yb)
        dW2 = (1 / m) * (A1.T @ dZ2)
        db2 = (1 / m) * np.sum(dZ2, axis=0, keepdims=True)

        #  hidden layer (ReLU)
        dZ1 = (dZ2 @ W2.T) * (Z1 > 0)
        dW1 = (1 / m) * (Xb.T @ dZ1)
        db1 = (1 / m) * np.sum(dZ1, axis=0, keepdims=True)

        # update
        W1 = W1 - lr * dW1
        b1 = b1 - lr * db1
        W2 = W2 - lr * dW2
        b2 = b2 - lr * db2

    if epoch % 200 == 0:
        # train loss (full)
        _, _, _, A2_tr = forward_relu_hidden(X_train, W1, b1, W2, b2)
        A2_tr = np.clip(A2_tr, eps, 1 - eps)
        tr_loss = -np.mean(y_train * np.log(A2_tr) + (1 - y_train) * np.log(1 - A2_tr))

        # test loss (full)
        _, _, _, A2_te = forward_relu_hidden(X_test, W1, b1, W2, b2)
        A2_te = np.clip(A2_te, eps, 1 - eps)
        te_loss = -np.mean(y_test * np.log(A2_te) + (1 - y_test) * np.log(1 - A2_te))

        train_loss_hist.append(tr_loss)
        test_loss_hist.append(te_loss)

        print(f"Epoch {epoch:4d} | train_loss={tr_loss:.4f} | test_loss={te_loss:.4f}")

def predict(X):
    Z1 = X @ W1 + b1
    A1 = np.maximum(0, Z1)
    Z2 = A1 @ W2 + b2
    return (Z2 > 0.5).astype(int)

def compute_accuracy(y_true, y_pred):
